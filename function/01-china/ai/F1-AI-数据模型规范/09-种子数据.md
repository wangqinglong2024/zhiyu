## 2. 文章 / 句子（dev 演示数据）

> 按 PM 答 Q3，所有翻译已"完整翻译好"，作为后续直接导入数据库的样例。
> 按 PM 答 Q12，每条句子都带 `pinyin` 字段。

```sql
-- 示例文章 1：秦始皇统一六国（已发布）
with new_article as (
  insert into zhiyu.china_articles (code, category_id, title_pinyin, title_i18n, status, published_at)
  select
    zhiyu.fn_gen_article_code(),
    c.id,
    'qín shǐ huáng tǒng yī liù guó',
    '{"zh":"秦始皇统一六国","en":"Qin Shi Huang Unifies the Six Kingdoms","vi":"Tần Thủy Hoàng thống nhất sáu nước","th":"จิ๋นซีฮ่องเต้รวมหกแคว้น","id":"Qin Shi Huang Menyatukan Enam Kerajaan"}'::jsonb,
    'published',
    now()
  from zhiyu.china_categories c where c.code = '01'
  returning id
)
insert into zhiyu.china_sentences
  (article_id, seq_no, pinyin, content_zh, content_en, content_vi, content_th, content_id, audio_status)
select a.id, v.seq_no, v.pinyin, v.zh, v.en, v.vi, v.th, v.id, 'pending'
from new_article a, (values
  (1,
   'gōng yuán qián 221 nián , qín wáng yíng zhèng wán chéng tǒng yī dà yè .',
   '公元前221年，秦王嬴政完成统一大业。',
   'In 221 BCE, King Ying Zheng of Qin completed the unification.',
   'Năm 221 TCN, Tần Vương Doanh Chính hoàn thành đại nghiệp thống nhất.',

-- 示例文章 2：宫保鸡丁（草稿）
with new_article as (
  insert into zhiyu.china_articles (code, category_id, title_pinyin, title_i18n, status)
  select
    zhiyu.fn_gen_article_code(),
    c.id,
    'gōng bǎo jī dīng',
    '{"zh":"宫保鸡丁","en":"Kung Pao Chicken","vi":"Gà Cung Bảo","th":"ไก่ผัดพริกกังเปา","id":"Ayam Kung Pao"}'::jsonb,
    'draft'
  from zhiyu.china_categories c where c.code = '02'
  returning id
)
insert into zhiyu.china_sentences
  (article_id, seq_no, pinyin, content_zh, content_en, content_vi, content_th, content_id, audio_status)
select a.id, v.seq_no, v.pinyin, v.zh, v.en, v.vi, v.th, v.id, 'pending'
from new_article a, (values
  (1,
   'gōng bǎo jī dīng shì chuān cài de jīng diǎn dài biǎo .',
   '宫保鸡丁是川菜的经典代表。',
   'Kung Pao Chicken is a classic dish of Sichuan cuisine.',
   'Gà Cung Bảo là món tiêu biểu của ẩm thực Tứ Xuyên.',
   'ไก่ผัดพริกกังเปาเป็นเมนูคลาสสิกของอาหารเสฉวน.',
   'Ayam Kung Pao adalah hidangan klasik masakan Sichuan.'),
  (2,
   'zhǔ liào shì jī ròu 、 huā shēng hé gān là jiāo .',
   '主料是鸡肉、花生和干辣椒。',
   'The main ingredients are chicken, peanuts, and dried chilies.',
   'Nguyên liệu chính là thịt gà, đậu phộng và ớt khô.',
   'วัตถุดิบหลักคือไก่ ถั่วลิสง และพริกแห้ง.',
   'Bahan utamanya adalah ayam, kacang, dan cabai kering.')
) as v(seq_no, pinyin, zh, en, vi, th, id);
```

> 覆盖：1 篇 `published` + 1 篇 `draft`，每篇多句，含 5 语言完整翻译 + 拼音；音频均 `pending`，等待用户首次点击触发 TTS（PM 答 Q10）。
